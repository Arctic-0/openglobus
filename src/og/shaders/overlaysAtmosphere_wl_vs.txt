const float C = 0.1;
const float far = 149.6e+9;
float logc = 2.0 / log( C * far + 1.0 );

attribute vec3 aVertexNormal;
attribute vec3 aVertexPosition;
attribute vec2 aTextureCoord;

uniform mat4 projectionMatrix;
uniform mat4 viewMatrix;
uniform mat3 normalMatrix;
uniform vec3 uNormalMapBias;
//uniform vec3 geoImageTexBias;
uniform vec4 uGlobalTextureCoord;

varying vec4 vTextureCoord;
varying vec3 vNormal;
varying vec4 vPosition;
varying mat3 v_NMatrix;
//varying vec2 vTextureCoordClear;
varying vec2 vGlobalTextureCoord;

void main(void) {
    vPosition = viewMatrix * vec4(aVertexPosition, 1.0);
    gl_Position = projectionMatrix * vPosition;
    gl_Position.z = ( log( C * gl_Position.w + 1.0 ) * logc - 1.0 ) * gl_Position.w;
    vNormal = aVertexNormal;
    v_NMatrix = normalMatrix;
    vTextureCoord.xy = aTextureCoord;
    vTextureCoord.zw = uNormalMapBias.z * ( aTextureCoord + vec2(uNormalMapBias.x, uNormalMapBias.y) );
	//vTextureCoordClear = geoImageTexBias.z *( aTextureCoord + vec2(geoImageTexBias.x, geoImageTexBias.y) );
	vGlobalTextureCoord = uGlobalTextureCoord.xy + (uGlobalTextureCoord.zw - uGlobalTextureCoord.xy) * aTextureCoord;
}
